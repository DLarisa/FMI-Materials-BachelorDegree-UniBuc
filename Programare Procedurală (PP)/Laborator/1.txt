#include <stdio.h>
#include <stdlib.h>

int main()
{
    float x, y, s=0;
    int nr=1, zile=1;
    scanf("%f%f",&x,&y);
    while(y!=0)
    {
        s=s+y;zile++;
        if(s<x) nr++;
        scanf("%f",&y);
    }
    printf("%d %.2f",nr,s/zile);

    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main()
{
    int nr1, nr2, x;
    scanf("%d",&x); nr1=nr2=x;
    while(x!=0)
    {
        if(x>nr1) nr2=nr1, nr1=x;
        else if(x>nr2 && x!=nr1) nr2=x;
        scanf("%d",&x);
    }
    printf("%d %d",nr1,nr2);

    return 0;
}
