#include <stdio.h>
#include <stdlib.h>
#include "cod.h"  //Ti le definesti singur

int main()
{
    int a,b;
    scanf("%d%d", &a, &b);
    char o; scanf("\n%c", &o);
    switch(o)
    {
        case '+': printf("%d\n",a+b); break;
        case '-': printf("%d\n",a-b); break;
        case '*': printf("%d\n",a*b); break;
        case '%': printf("%d\n",a%b); break;
        case '/':
            if(b!=0) printf("%d\n",a/b);
            else printf("b=0 ---> Imposibil de Executat.\n"); break;
        default: printf("Nu a fost oferit un operator valid.\n"); break;
    }


    return 0;
}
