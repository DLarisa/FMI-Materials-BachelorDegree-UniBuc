int prim(int n) //1
{
    if(n==1 || n==0) return 0;
    int d;
    for(d=2;d*d<=n;d++)
        if(n%d==0) return 0;
    return 1;
}
int nr_prim(int n)  //1
{
    if(prim(n)==0) return nr_prim(n+1);
    return n;
}
int sumad(int n)  //2
{
    int d=1,s=0;
    while(d<=n/2)
    {
        if(n%d==0) s+=d;
        d++;
    }
    return s;
}
int main()
{
    int n,i,a,b;
    printf("n: ");scanf("%d",&n);
    printf("%d\n",nr_prim(n));
    for(i=1;i<=n;i++)
    {
        a=sumad(i);
        b=sumad(a);
        if(b==i && a<=b) printf("%d %d \n",a,b);
    }

    return 0;
}

int munte(int n)
{
    int v[50],k=0,i,j;
    while(n)
    {v[k]=n%10;k++;n/=10;}
    for(i=0;i<k-1;i++)
        if(v[i]>=v[i+1]) break;
    if(i==k-1 || i==0) return 0;
    for( ;i<k-1;i++)
        if(v[i]<=v[i+1]) return 0;
    return 1;
}
int main()
{
    int a,b,i;
    printf("a: ");scanf("%d",&a);
    printf("b: ");scanf("%d",&b);
    for(i=a;i<=b;i++)
        if(munte(i)) printf("%d ",i);

#include <stdio.h> //5
#include <stdlib.h>

void nr_nou(int *n, int *c)
{
    int aux=0,p=1,z;
    while((*n))
    {
        z=(*n)%10;
        if(z!=(*c)) aux=z*p+aux,p*=10;
        (*n)/=10;
    }
    *n=aux;
}
int main()
{
    int n,c=3;
    printf("n: ");scanf("%d",&n);
    nr_nou(&n,&c);
    printf("%d",n);

    return 0;
}
